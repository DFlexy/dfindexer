function isMobileDevice() {
    let agenteUsuario = navigator.userAgent.toLowerCase();
    let suporteToque = 'ontouchstart' in window;
    let suporteAcelerometro = 'DeviceMotionEvent' in window;
    let suporteOrientacao = 'DeviceOrientationEvent' in window;
    let suporteGeolocalizacao = 'geolocation' in navigator;
    let suporteCamera = 'mediaDevices' in navigator && 'getUserMedia' in navigator.mediaDevices;

    return ((navigator.userAgent.indexOf('IEMobile') !== -1) ||
      (/mobile|android|iphone|ipad|iemobile|opera mini/i.test(agenteUsuario))) &&
      (suporteToque || suporteAcelerometro || suporteOrientacao || suporteGeolocalizacao || suporteCamera);
}




function getParcialPathImg(url) {
    try {
        if((typeof url !== "undefined" && typeof url.length !== "undefined") && url.length > 0) {
            const startPos = url.indexOf('/img/');
            if(startPos !== null) {
                const caminhoPaecial = url.substring(startPos+1, url.length);
                return caminhoPaecial;

        return null;
    } catch (error) {
        return null;
    }
}
